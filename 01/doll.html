코드

function solution(new_id) {
    var answer = '';
    new_id = new_id.toLowerCase(); // 1
    
    // 2
    let pattern = /[a-zA-Z]/; //영어
    let pattern2 = /[0-9]/;
    let cond2 = '';
    for(let n of new_id){  // 한글자씩 비교하면서 조건에 맞는 문자들만 추가
        if(n === '.' || n === '_' || n === '-' || pattern.test(n) || pattern2.test(n)){
            cond2 += n
        }
    }
    // 3
    let cond3 = '';
    let prev = '';
    for(let c of cond2){
        if(c === prev && c === '.') continue; // .이 연속으로 나왔다면
        cond3 += c;
        prev = c;
    }

    // 4
    if(cond3.charAt(0) === '.') cond3 = cond3.slice(1);
    if(cond3.charAt(cond3.length-1) === '.') cond3 = cond3.slice(0,-1);
    
    // 5
    if(cond3 === ''){
        cond3 = 'a';
    }
    
    // 6
    if(cond3.length > 15){
        cond3 = cond3.slice(0,15);
        if(cond3.charAt(cond3.length-1) === '.') cond3 = cond3.slice(0,-1);
    }
    
    // 7
    while(cond3.length < 3){
        cond3 += cond3.charAt(cond3.length-1);
    }
    answer = cond3;
    return answer;
}

1. 1~7번 까지 조건에 맞춰서 처리해 주면 된다.

총평 : 정규식 체이닝을 이용해서 간단하게 풀 수 있는것 같다. 2번조건에서도 정규식을 사용하지 않는다면 난감한 경우가 있고, 정규식을 알고 모르고가 코드에 질이 달라지는 것 같다.
