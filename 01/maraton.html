1 단계 : 완주하지 못한 선수

코딩테스트 연습 > 해시 > 완주하지 못한 선수

코드

function solution(participant, completion) {
    participant.sort();
    completion.sort(); // 정렬을 해줘야 순서를 비교할 수 있다.
    
    for(let p in participant){ // 참가자를 순회하면서
        if(participant[p] !== completion[p]) return participant[p]; 
        // 참가자 배열의 인덱스와 다른 completion 배열의 원소가 있으면 그 때의 원소가 완주하지 못한 선수이다.
    }
    return participant[participant.length-1]; // 만약 끝까지 돌았는데 못찾았으면 마지막 선수가 완주하지 못한 선수이다.
    // for(let i in participant) {
    //     if(completion.includes(participant[i]) == false) return participant[i];
    //     completion.splice(completion.indexOf(participant[i]), 1);
    // }
}

리뷰

순서는 다음과 같다
1. 배열의 순서를 비교하기 위해 두 배열을 정렬한다.
2. 참가자 배열을 순회하면서 참가자 배열과 완주자 배열의 원소가 같은 것끼리 비교해서 다른게 있다면 그 참가자가 완주하지 못한 선수이다.
3. 만약 끝까지 돌았는데 못찾았다면 마지막 참가자가 완주하지 못한 선수이다.
